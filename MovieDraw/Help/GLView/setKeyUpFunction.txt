<desc>Sets a user function that is called every time a key is released.</desc>
<example>
void function(NSEvent* event)
{
	// Does something if the W key is released
	unsigned short key = [ [ event characters ] characterAtIndex:0 ];
	if (key == 'w' || key == 'W')
		DoSomething();
}
GLView* view = //...
// Sets the function
[ view setKeyUpFunction:function ];
</example>
