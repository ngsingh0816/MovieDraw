<desc>Sets the key up function to the one supplied. This function is called every time a new key is released.</desc>
<example>
// Sets the function
void function(NSEvent* event)
{
	// Check if 'W' was released
	unsigned short key = [ [ event characters ] characterAtIndex:0 ];
	if (key == 'w' || key == 'W')
	{
		// 'W' was released, do something
		DoSomething();
	}
}
MDSetKeyUpFunction(function);
</example>
