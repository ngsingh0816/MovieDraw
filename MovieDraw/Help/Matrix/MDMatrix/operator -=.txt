<name>operator -=</name>
<type>
	<name>Function</name>
	<return>
		<type>MDMatrix</type>
		<desc>Returns the two matrices subtracted from each other.</desc>
	</return>
	<argument0>
		<type>MDMatrix</type>
		<name>p</name>
		<desc>The matrix to subtract from the other matrix.</desc>
	</argument0>
</type>
<desc>Subtracts two matrices from each other and sets the current matrix to the difference.</desc>
<example>
MDMatrix matrix = MDMatrixIdentity();
MDMatrix matrix2 = MDMatrixIdentity();
matrix -= matrix3;		// Gives a new matrix with all zeroes
</example>
