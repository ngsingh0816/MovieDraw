</type>
<desc>Creates and returns a new instance of a MDPoint that is a copy of the supplied point. A MDPoint created by this method should either be autoreleased or released when not needed anymore.</desc>
<example>
// Initializes a point at (3, 0, -2) with the normal (0, 1, 0)
MDPoint* point = [ [ MDPoint alloc ] init ];
[ point setPosition:MDVector3Create(3, 0, -2) ];
[ point setNormal:MDVector3Create(0, 1, 0) ];
// Create a copy of that point
MDPoint* point2 = [ [ MDPoint alloc ] initWithPoint:point ];
// Cleanup
[ point release ];
</example>
